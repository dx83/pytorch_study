# 파이토치를 이용한 합성곱 신경망 구현

## 배치 정규화 적용 합성곱 신경망
- 배치 정규화는 각 미니 배치의 활성화값을 정규화해 학습 속도를 높이고 과적합을 방지하는 기법이다.

```python
class CNNWithBatchNorm(nn.Module):
    def __init__(self):
        super(CNNWithBatchNorm, self).__init__()
        # 첫 번째 합성곱
        self.conv1 = nn.Conv2d(3, 16, kernel_size=3, stride=1, padding=1)
        self.bn1 = nn.BatchNorm2d(16)   # 배치 정규화 추가
        self.relu1 = nn.ReLU()
        self.pool1 = nn.MaxPool2d(kernel_size=2, stride=2)
        # 두 번째 합성곱
        self.conv2 = nn.Conv2d(16, 32, kernel_size=3, stride=1, padding=1)
        self.bn2 = nn.BatchNorm2d(32)   # 배치 정규화 추가
        self.relu2 = nn.ReLU()
        self.pool2 = nn.MaxPool2d(kernel_size=2, stride=2)
        # 완전 연결 레이어
        self.fc = nn.Linear(2048, 10)

    def forward(self, x):
        # 첫 번째 블록
        x = self.conv1(x)
        x = self.bn1(x)     # 배치 정규화 적용
        x = self.relu1(x)
        x = self.pool1(x)
        # 두 번째 블록
        x = self.conv2(x)
        x = self.bn2(x)     # 배치 정규화 적용
        x = self.relu2(x)
        x = self.pool2(x)
        # 평탄화
        x = x.view(x.size(0), -1)
        # 완전 연결 레이어
        x = self.fc(x)
        return x

# 모델 생성 및 테스트
model = CNNWithBatchNorm()
print(model)
# 더미 입력으로 테스트
dummy_input = torch.randn(1, 3, 32, 32)
output = model(dummy_input)
print(f"출력 크기: {output.shape}")
```

- 배치 정규화의 주요 효과
    - 학습 속도 향상
    - 초기화에 대한 의존성 감소
    - 정규화 효과
    - 더 높은 학습률 사용 가능
