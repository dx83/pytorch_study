# 파이토치를 이용한 합성곱 신경망 구현

## 드롭아웃 (Dropout)을 통한 과적합 방지
- 드롭아웃은 학습 과정에서 임의의 뉴런을 일정 확률로 비활성화하여 과적합을 방지하는 정규화 기법이다.

```python
import torch
import torch.nn as nn
# 드롭아웃 적용 합성곱 신경망
class CNNWithDropout(nn.Module):
    def __init__(self):
        super(CNNWithDropout, self).__init__()
        # 첫 번째 합성곱
        self.conv1 = nn.Conv2d(in_channels=3, out_channels=16, kernel_size=3, padding=1)
        self.bn1 = nn.BatchNorm2d(16)
        self.relu1 = nn.ReLU()
        self.pool1 = nn.MaxPool2d(kernel_size=2, stride=2)
        self.dropout1 = nn.Dropout2d(0.25)  # 25% 드롭아웃 적용
        # 두 번째 합성곱
        self.conv2 = nn.Conv2d(in_channels=16, out_channels=32, kernel_size=3, padding=1)
        self.bn2 = nn.BatchNorm2d(32)
        self.relu2 = nn.ReLU()
        self.pool2 = nn.MaxPool2d(kernel_size=2, stride=2)
        self.dropout2 = nn.Dropout2d(0.25)  # 25% 드롭아웃 적용
        # 완전 연결 레이어
        self.fc1 = nn.Linear(32 * 8 * 8, 128)
        self.dropout3 = nn.Dropout2d(0.5)   # 50% 드롭아웃 적용
        self.fc2 = nn.Linear(128, 10)

    def forward(self, x):
        # 첫 번째 블록
        x = self.conv1(x)
        x = self.bn1(x)
        x = self.relu1(x)
        x = self.pool1(x)
        x = self.dropout1(x)    # 드롭아웃 적용
        # 두 번째 블록
        x = self.conv2(x)
        x = self.bn2(x)
        x = self.relu2(x)
        x = self.pool2(x)
        x = self.dropout2(x)    # 드롭아웃 적용
        # 평탄화
        x = x.view(x.size(0), -1)
        # 완전 연결 레이어
        x = self.fc1(x)
        x = self.dropout3(x)    # 드롭아웃 적용
        x = self.fc2(x)
        return x

# 모델 생성 및 출력
model = CNNWithDropout()
print(model)
```
