# 파이토치를 이용한 합성곱 신경망 구현

## 기본 합성곱 신경망 아키텍처 설계
- Conv2d 와 MaxPool2d 레이어 사용법

```python
import torch
import torch.nn as nn

class SimpleCNN(nn.Module):
    def __init__(self):
        super(SimpleCNN, self).__init__()
        # 첫 번째 합성곱
        self.conv1 = nn.Conv2d(in_channels=3, out_channels=16, kernel_size=3, stride=1, padding=1)
        self.relu1 = nn.ReLU()
        self.pool1 = nn.MaxPool2d(kernel_size=2, stride=2)
        # 두 번째 합성곱
        self.conv2 = nn.Conv2d(in_channels=16, out_channels=32, kernel_size=3, stride=1, padding=1)
        self.relu2 = nn.ReLU()
        self.pool2 = nn.MaxPool2d(kernel_size=2, stride=2)
        # 완전 연결 레이어
        self.fc = nn.Linear(2048, 10)

    def forward(self, x):
        # 첫 번째 블록             입력 3 x 32 x 32
        x = self.conv1(x)          # 16 x 32 x 32
        x = self.relu1(x)
        x = self.pool1(x)          # 16 x 16 x 16
        # 두 전째 블록
        x = self.conv2(x)          # 32 x 16 x 16
        x = self.relu2(x)
        x = self.pool2(x)          # 32 x  8 x  8
        # 평탄화
        x = x.view(x.size(0), -1)  # 32 x  8 x  8 = 2048
        # 완전 연결 레이어
        x = self.fc(x)             # 출력 10
        return x

# 모델 인스턴스 생성
model = SimpleCNN()
print(model)
# 임의의 입력으로 모델 테스트
sample_input = torch.randn(1, 3, 32, 32)
output = model(sample_input)
print(f"출력 크기: {output.shape}")  # [1, 10]
```
`nn.Conv2d`
- in_channels= 입력 특성 맵의 채널 수
- out_channels= 출력 특성 맵의 채널 수 (필터 수)
- kernel_size= 커널(필터)의 크기
- stride= 스트라이드 값
- padding= 패딩 크기

`nn.MaxPool2d`
- kernel_size= 풀링 윈도우의 크기
- stride= 스트라이드값 (기본값은 kernel_size와 동일)

`합성곱 출력 높이/너비 계산`
$$
출력 = \left\lfloor \frac{입력 + 2 \times 패딩 - 커널}{스트라이드} \right\rfloor + 1
$$
